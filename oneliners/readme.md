### Drip in 6502 machine language

This is a direct machine-language version of the Drip loop (using `x=0..7` as in your variant):

```asm
; C64 6502
; Build to $C000 and run with: SYS 49152

* = $C000

S       = $C080
TLO     = $C081
THI     = $C082
SCRLO   = $FB
SCRHI   = $FC
COLLO   = $FD
COLHI   = $FE

START:
        LDA #$00
        STA TLO
        STA THI
        LDX #$08                ; FOR X=0 TO 7

LOOP:
        ; SCRPTR = $0400 + T
        CLC
        LDA TLO
        ADC #<$0400
        STA SCRLO
        LDA THI
        ADC #>$0400
        STA SCRHI

        ; COLPTR = $D800 + T
        CLC
        LDA TLO
        ADC #<$D800
        STA COLLO
        LDA THI
        ADC #>$D800
        STA COLHI

        LDY #$00
        LDA #$0E                ; color 14
        STA (COLLO),Y

        LDA (SCRLO),Y
        ORA #$80                ; PEEK(...) OR 128
        STA (SCRLO),Y

        ; T = (T + S) AND 1023
        CLC
        LDA TLO
        ADC S
        STA TLO
        LDA THI
        ADC #$00
        AND #$03
        STA THI

        DEX
        BNE LOOP

        INC S                   ; S=S+1
        JMP START               ; GOTO

        ; state byte (defaults to 0)
        * = S
        .BYTE $00,$00,$00
```

60 INPUT"RANDOM START (0=NO,1=YES)";RS

80 REM INIT ROW
90 FOR I=0 TO 39:A(I)=0:NEXT
100 IF RS=1 THEN GOTO 140
110 A(19)=1:GOTO 170
140 FOR I=0 TO 39
150 A(I)= (RND(1)>.5)
160 NEXT

180 FOR Y=0 TO 24
190 REM DRAW CURRENT ROW
200 FOR X=0 TO 39
210 P=1024+Y*40+X
220 IF A(X)=1 THEN POKEP,81:POKE55296+Y*40+X,1:GOTO 240
230 POKEP,32:POKE55296+Y*40+X,0

250 REM COMPUTE NEXT ROW USING RULE R
260 FOR X=0 TO 39
270 L=A((X+39)AND39):C=A(X):RR=A((X+1)AND39)
280 N=L*4+C*2+RR
290 REM BIT (7-N) OF R DECIDES NEXT STATE
300 B(X)=((INT(R/2^(7-N))AND1)<>0)
310 NEXT X

320 REM COPY NEXT->CURRENT
330 FOR X=0 TO 39:A(X)=B(X):NEXT

340 NEXT Y

350 PRINT:INPUT"AGAIN (Y/N)";Q$
360 IF Q$="Y" OR Q$="y" THEN 20
370 END
